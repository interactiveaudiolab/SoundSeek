260  vector<path> getNearestWeighted (path query, int num_results = 5)
261  {
262  addFile (query);
263 
264  // auto query_id = pathToID (query);
265  clock_t start = clock ();
266  vector<double> dists (sounds.size ());
267  for (int i = 0; i < sounds.size (); ++i)
268  {
269  if (sounds[i] == query)
270  dists[i] = numeric_limits<double>::max ();
271  else
272  dists[i] = getWeightedDistance (query, sounds[i]);
273  }
274 
275  // argsort
276  vector<int> indices (sounds.size ());
277  iota (indices.begin (), indices.end (), 0);
278 
279  sort (indices.begin (), indices.end (), [&](int a, int b) { return dists[a] < dists[b]; });
280 
281  indices.pop_back ();
282 
283  vector<path> results;
284 
285  for (int i = 0; i < num_results; ++i)
286  {
287  if (sounds[indices[i]] != query) results.push_back (sounds[indices[i]]);
288  }
289  double duration = (clock () - start) / (double) CLOCKS_PER_SEC;
290  cerr << "Searched " << sounds.size () << " sounds in " << duration << " seconds" << endl;
291
